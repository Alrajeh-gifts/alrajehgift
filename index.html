// --- ⏰ المؤقت المدمج الذكي (مرتبط بقفل زر التسليم) --- function initCountdown() { const eidDate = new Date("2026-03-20T00:00:00").getTime(); setInterval(() => { const now = new Date().getTime(); const diff = eidDate - now; const timerEl = document.getElementById('countdown-timer'); const deliverBtn = document.getElementById('deliver-btn'); // جلب زر التسليم if (!timerEl) return; if (diff < 0) { // إذا جاء العيد: افتح الزر timerEl.innerText = "🎉 العيد بدأ! يمكنك التسليم الآن"; if (deliverBtn) { deliverBtn.disabled = false; deliverBtn.style.opacity = '1'; deliverBtn.innerHTML = ` 📦 تأكيد تسليم الهدية اضغط هنا لتأكيد التسليم `; } return; } // إذا لم يأتِ العيد بعد: اقفل الزر const d = Math.floor(diff / (1000 * 60 * 60 * 24)); const h = Math.floor((diff % (1000 * 60 * 60 * 24)) / (1000 * 60 * 60)); timerEl.innerText = `⏳ باقي: ${d} يوم و ${h} ساعة`; if (deliverBtn && !deliverBtn.classList.contains('hidden-el')) { deliverBtn.disabled = true; // تعطيل الضغط deliverBtn.style.opacity = '0.5'; // جعله باهتاً ليعرف المستخدم أنه مقفل deliverBtn.innerHTML = ` 🔒 التسليم مغلق يفتح يوم العيد `; } }, 1000); } initCountdown();
